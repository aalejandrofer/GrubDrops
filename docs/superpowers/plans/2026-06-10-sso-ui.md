# SSO UI (login redesign + settings card) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make SSO the primary login CTA (generic "Continue with SSO" button, password kept secondary), and add a read-only SSO status card to the General settings page.

**Architecture:** Pure presentation change on the existing OIDC feature. The login template gains a two-state layout (SSO-primary when enabled, password-only when disabled). `oidc.Provider` gains read-only getters so the settings page can show issuer/redirect/allowlist status. No new env, no DB, no change to the auth flow or handlers.

**Tech Stack:** Go html/template (server-rendered), hand-rolled CSS in a single `app.css` (dark theme, CSS-var tokens), `coreos/go-oidc`, testify. Tests render templates via `web.Templates()` and assert substrings.

Spec: `docs/superpowers/specs/2026-06-10-sso-ui-design.md`

func (p *Provider) AllowedEmails() []string { return p.allowedEmails }

func (p *Provider) AllowedGroups() []string { return p.allowedGroups }

func renderLogin(t *testing.T, data templateData) string {
	t.Helper()
	tmpl, err := web.Templates()
	if err != nil {
		t.Fatalf("load templates: %v", err)
	}
	var buf bytes.Buffer
	if err := tmpl.ExecuteTemplate(&buf, "login.html", data); err != nil {
		t.Fatalf("render login: %v", err)
	}
	return buf.String()
}

func TestLogin_SSOEnabled_ShowsPrimaryButton(t *testing.T) {
	out := renderLogin(t, templateData{OIDCEnabled: true, OIDCProviderName: "authentik"})
	if !strings.Contains(out, "Continue with SSO") {
		t.Errorf("expected generic SSO button label, got:\n%s", out)

		t.Errorf("expected SSO link to /auth/oidc/login")

		t.Errorf("login button must NOT name the provider")
	}

		t.Errorf("SSO button must be hidden when OIDC disabled")
	}
	if !strings.Contains(out, `name="password"`) {
		t.Errorf("expected password field present")
	}
}
